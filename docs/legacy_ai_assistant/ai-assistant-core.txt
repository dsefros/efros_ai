# providers/base_provider.py
from abc import ABC, abstractmethod
from typing import Dict, List, Any, Optional

class BaseProvider(ABC):
    """Базовый интерфейс для всех провайдеров генерации ответов"""
    
    @abstractmethod
    def generate_answer(self, query: str, context: Optional[Dict] = None) -> str:
        """Генерирует ответ на основе запроса и контекста"""
        pass
    
    @abstractmethod
    def get_provider_name(self) -> str:
        """Возвращает имя провайдера для логирования и маршрутизации"""
        pass
    
    def get_context(self, query: str) -> Optional[Dict]:
        """Получает контекст для провайдера (опционально, для RAG)"""
        return None
    
    def get_sources(self, context: Dict) -> List[Dict]:
        """Извлекает источники из контекста (для RAG)"""
        return []

# providers/provider_factory.py
import os
from .base_provider import BaseProvider
from .rag_provider import RAGProvider
from .yandex_provider import YandexProvider
from .gigachat_provider import GigaChatProvider
import logging

class ProviderFactory:
    """Фабрика для создания провайдеров"""
    
    def __init__(self, config):
        self.config = config
        self.providers = {}
        self.logger = logging.getLogger(__name__)
        self._register_providers()
    
    def _register_providers(self):
        """Регистрирует все доступные провайдеры"""
        self.logger.info("🔧 Registering providers...")
        
        # RAG провайдер (всегда доступен если Qdrant работает)
        try:
            self.providers['rag'] = RAGProvider(self.config)
            self.logger.info("✅ RAG provider registered")
        except Exception as e:
            self.logger.error(f"❌ Failed to register RAG provider: {e}")
        
        # YandexGPT провайдер (если настроен в config)
        if self.config.ai_provider == 'yandexgpt' or 'YANDEX_API_KEY' in os.environ:
            try:
                self.providers['yandex'] = YandexProvider(self.config)
                self.logger.info("✅ Yandex provider registered")
            except Exception as e:
                self.logger.error(f"❌ Failed to register Yandex provider: {e}")
        
        # GigaChat провайдер (если настроен в config)
        if self.config.ai_provider == 'gigachat' or 'GIGACHAT_API_KEY' in os.environ:
            try:
                self.providers['gigachat'] = GigaChatProvider(self.config)
                self.logger.info("✅ GigaChat provider registered")
            except Exception as e:
                self.logger.error(f"❌ Failed to register GigaChat provider: {e}")
    
    def get_provider(self, provider_name: str) -> BaseProvider:
        """Получает провайдера по имени"""
        if provider_name not in self.providers:
            available = ", ".join(self.providers.keys())
            raise ValueError(f"Provider '{provider_name}' not available. Available providers: {available}")
        
        return self.providers[provider_name]
    
    def get_default_provider(self, channel: str) -> BaseProvider:
        """Получает провайдера по умолчанию для канала"""
        # Конфигурация по умолчанию для разных каналов
        channel_defaults = {
            'redmine': 'rag',    # Для Redmine используем RAG по умолчанию
            'cli': 'rag',        # Для CLI используем RAG
            'telegram': 'yandex' # Для Telegram используем Yandex (пока)
        }
        
        provider_name = channel_defaults.get(channel, 'rag')
        return self.get_provider(provider_name)

# providers/rag_provider.py
from .base_provider import BaseProvider
from retrieval.vector_store import QdrantVectorStore
from retrieval.reranker import Reranker
from generation.llm_client import LLMClient
import logging
from typing import Dict, List, Any, Optional

class RAGProvider(BaseProvider):
    """Провайдер для локальной RAG-системы"""
    
    def __init__(self, config):
        self.logger = logging.getLogger(__name__)
        self.logger.info("🚀 Initializing RAG Provider...")
        
        # Инициализация компонентов RAG
        self.product_store = QdrantVectorStore(collection_name="rag_product")
        self.regulatory_store = QdrantVectorStore(collection_name="rag_regulatory")
        self.reranker = Reranker()
        # Передаем строковый путь к модели
        self.llm = LLMClient(model_path=config.model_path)
        
        self.logger.info("✅ RAG Provider initialized successfully")
    
    def get_provider_name(self) -> str:
        return "rag"
    
    def get_context(self, query: str) -> Dict[str, Any]:
        """Получает гибридный контекст из обеих баз"""
        self.logger.info(f"🔍 Getting context for query: {query[:100]}...")
        
        # Получаем релевантные документы из обеих коллекций
        product_docs = self.product_store.search(query, top_k=20)
        regulatory_docs = self.regulatory_store.search(query, top_k=20)
        
        # Объединяем и реранжируем
        all_docs = []
        for doc in product_docs:
            doc["source_db"] = "rag_product"
            all_docs.append(doc)
        
        for doc in regulatory_docs:
            doc["source_db"] = "rag_regulatory"
            all_docs.append(doc)
        
        # Применяем реранкинг
        reranked_docs = self.reranker.rerank(query, all_docs, top_k=40)
        
        # Реализуем логику 5+3 как в оригинальном app.py
        product_reranked = [r for r in reranked_docs if r["source_db"] == "rag_product"]
        regulatory_reranked = [r for r in reranked_docs if r["source_db"] == "rag_regulatory"]
        
        final_context = []
        # Берём максимум 5 из продуктовой базы
        product_limit = min(5, len(product_reranked))
        final_context.extend(product_reranked[:product_limit])
        
        # Берём максимум 3 из регуляторной базы
        regulatory_limit = min(3, len(regulatory_reranked))
        final_context.extend(regulatory_reranked[:regulatory_limit])
        
        # Если набрали меньше 8, добираем из оставшихся
        remaining_slots = 8 - len(final_context)
        if remaining_slots > 0:
            other_results = []
            if len(product_reranked) > product_limit:
                other_results.extend(product_reranked[product_limit:])
            if len(regulatory_reranked) > regulatory_limit:
                other_results.extend(regulatory_reranked[regulatory_limit:])
            # Сортируем по релевантности и добираем
            other_results.sort(key=lambda x: x["rerank_score"], reverse=True)
            final_context.extend(other_results[:remaining_slots])
        
        self.logger.info(f"📚 Retrieved {len(final_context)} context documents "
                        f"({len([c for c in final_context if c['source_db']=='rag_product'])} продуктовых, "
                        f"{len([c for c in final_context if c['source_db']=='rag_regulatory'])} регуляторных)")
        return {
            "documents": final_context,
            "query": query
        }
    
    def get_sources(self, context: Dict[str, Any]) -> List[Dict[str, Any]]:
        """Извлекает источники из контекста"""
        if not context or "documents" not in context:
            return []
        
        return [
            {
                "source": doc["metadata"]["source"],
                "page": doc["metadata"].get("page_number", "N/A"),
                "source_db": doc["source_db"],
                "text_preview": doc["text"][:200] + "..." if len(doc["text"]) > 200 else doc["text"]
            }
            for doc in context["documents"]
        ]
    
    def generate_answer(self, query: str, context: Optional[Dict[str, Any]] = None) -> str:
        """Генерирует ответ с использованием RAG"""
        self.logger.info(f"🧠 Generating RAG answer for: {query[:100]}...")
        
        if context is None:
            context = self.get_context(query)
        
        context_docs = context["documents"]
        
        if not context_docs:
            return "❌ Не найдено релевантных документов для ответа на ваш вопрос."
        
        # Генерация ответа
        answer = self.llm.generate_answer(query, context_docs)
        self.logger.info(f"✅ RAG answer generated: {answer[:100]}...")
        
        return answer

# generation/llm_client.py
import logging
from typing import List, Dict, Any
from llama_cpp import Llama

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class LLMClient:
    def __init__(
        self,
        model_path: str = "models/mistral-7b-instruct-v0.2.Q4_K_M.gguf",
        n_ctx: int = 8192,
        n_gpu_layers: int = 40,
        n_threads: int = 8,
        temperature: float = 0.1,
        max_tokens: int = 512
    ):
        logger.info(f"Загрузка LLM: {model_path}")
        self.llm = Llama(
            model_path=model_path,
            n_ctx=n_ctx,
            n_gpu_layers=n_gpu_layers,
            n_threads=n_threads,
            verbose=False
        )
        self.temperature = temperature
        self.max_tokens = max_tokens

    def generate_answer(self, query: str, context_documents: List[Dict[str, Any]]) -> str:
        """Генерирует ответ на основе запроса и контекста."""
        
        # Формируем контекст из документов
        context_text = "\n".join([
            f"Документ {i+1} (источник: {doc['metadata']['source']}):\n{doc['text']}"
            for i, doc in enumerate(context_documents)
        ])
        
        prompt = f"""Ты — эксперт службы поддержки. 
ВОПРОС: {query}
Отвечай ТОЛЬКО на основе предоставленных документов ниже.
ДОКУМЕНТЫ:
{context_text}
ИНСТРУКЦИИ:
1. Если ответ нельзя вывести из документов — скажи: «В предоставленных материалах нет информации по этому вопросу».
2. НИКОГДА не придумывай детали.
3. Ссылайся на источник: укажи название документа (например, «согласно инструкции по виртуальной кассе»).
4. Отвечай кратко, по делу, на русском языке.
ОТВЕТ:"""
        
        # Логируем длину промпта
        prompt_tokens = len(self.llm.tokenize(prompt.encode("utf-8")))
        logger.info(f"Длина промпта: {prompt_tokens} токенов")
        
        if prompt_tokens > 7500:  # Оставляем место для ответа
            logger.warning("⚠️ Промпт близок к лимиту контекста!")
        
        response = self.llm(
            prompt,
            max_tokens=self.max_tokens,
            temperature=self.temperature,
            stop=["\n", "ВОПРОС:", "ИНСТРУКЦИИ:"],
            echo=False
        )
        
        answer = response["choices"][0]["text"].strip()
        return answer if answer else "Не удалось сгенерировать ответ."

import urllib3
from typing import Dict, List, Any, Optional
from redminelib import Redmine
from redminelib.exceptions import ResourceAttrError, AuthError, ServerError
import logging
from retry import retry
from datetime import datetime
import re

# Отключаем предупреждения SSL
urllib3.disable_warnings()

def _remove_emojis(text: str) -> str:
    """Удаляет эмодзи и другие не-BMP символы, которые могут вызывать ошибки в Redmine"""
    emoji_pattern = re.compile(
        "["
        "\U0001F600-\U0001F64F"  # emoticons
        "\U0001F300-\U0001F5FF"  # symbols & pictographs
        "\U0001F680-\U0001F6FF"  # transport & map symbols
        "\U0001F1E0-\U0001F1FF"  # flags
        "\U00002700-\U000027BF"  # dingbats
        "]+",
        flags=re.UNICODE
    )
    return emoji_pattern.sub(r'', text)

class RedmineClient:
    """Клиент для работы с Redmine API (рабочая версия с redminelib)"""
    
    def __init__(self, redmine_url: str, api_key: str):
        self.redmine_url = redmine_url
        self.api_key = api_key
        self._redmine = None
        self.logger = logging.getLogger(__name__)
        self.logger.info(f"✅ Initialized Redmine client for {redmine_url}")
    
    @property
    def redmine(self):
        """Ленивая инициализация Redmine клиента"""
        if self._redmine is None:
            self._redmine = Redmine(
                self.redmine_url,
                key=self.api_key,
                requests={'verify': False}  # Критически важно для SSL
            )
        return self._redmine
    
    @retry(tries=3, delay=2, backoff=2, logger=logging)
    def find_target_issues(self, target_status_id: str) -> List[Dict]:
        """Находит задачи с целевым статусом"""
        self.logger.info(f"🔍 Searching for issues with status ID: {target_status_id}")
        try:
            # Используем filter вместо напрямую GET запроса
            issues = self.redmine.issue.filter(
                status_id=target_status_id,
                limit=100
            )
            result = []
            for issue in issues:
                result.append(self._format_issue(issue))
            self.logger.info(f"✅ Found {len(result)} issues with status ID {target_status_id}")
            return result
        except (AuthError, ServerError) as e:
            self.logger.error(f"❌ Redmine API error: {str(e)}")
            raise
        except Exception as e:
            self.logger.error(f"❌ Unexpected error searching issues: {str(e)}", exc_info=True)
            return []
    
    def get_issue_details(self, issue_id: int) -> Dict:
        """Получает детали задачи"""
        try:
            issue = self.redmine.issue.get(issue_id, include='attachments')
            return self._format_issue(issue)
        except Exception as e:
            self.logger.error(f"❌ Error getting issue details for #{issue_id}: {str(e)}")
            return {}
    
    def get_issue_display_info(self, issue: Dict) -> str:
        """Возвращает строку с информацией о задаче для логирования"""
        return f"#{issue.get('id', '?')} '{issue.get('subject', 'No subject')}' by {issue.get('author_name', 'Unknown')}"
    
    @retry(tries=3, delay=1, backoff=2, logger=logging)
    def add_analysis_comment(self, issue_id: int, analysis_text: str, documents_count: int = 0) -> bool:
        """Добавляет приватный комментарий с анализом к задаче"""
        comment = self._format_comment(analysis_text, documents_count)
        return self.add_private_comment(issue_id, comment)
    
    @retry(tries=3, delay=1, backoff=2, logger=logging)
    def add_private_comment(self, issue_id: int, comment: str) -> bool:
        """Добавляет приватный комментарий к задаче (рабочая версия)"""
        try:
            self.logger.info(f"💭 Adding private comment to issue #{issue_id}")
            
            # КРИТИЧЕСКИ ВАЖНО: сначала получаем объект задачи
            issue = self.redmine.issue.get(issue_id)
            
            # Обновляем только нужные поля
            issue.notes = comment
            issue.private_notes = True  # Делаем комментарий приватным
            
            # Сохраняем изменения - redminelib сам формирует правильный запрос
            issue.save()
            
            self.logger.info(f"✅ Private comment successfully added to issue #{issue_id}")
            return True
        except Exception as e:
            self.logger.error(f"❌ Error adding comment to issue #{issue_id}: {str(e)}")
            return False
    
    def _format_issue(self, issue) -> Dict:
        """Форматирует задачу в словарь"""
        try:
            return {
                "id": issue.id,
                "subject": self._safe_get_attr(issue, 'subject', "No title"),
                "description": self._safe_get_attr(issue, 'description', ""),
                "status": {
                    "id": self._safe_get_attr(issue.status, 'id', 0),
                    "name": self._safe_get_attr(issue.status, 'name', "Unknown")
                },
                "author": {
                    "id": self._safe_get_attr(issue.author, 'id', 0),
                    "name": self._safe_get_attr(issue.author, 'name', "Unknown")
                },
                "author_name": self._safe_get_attr(issue.author, 'name', "Unknown"),
                "assigned_to": {
                    "id": self._safe_get_attr(issue.assigned_to, 'id', 0) if hasattr(issue, 'assigned_to') else 0,
                    "name": self._safe_get_attr(issue.assigned_to, 'name', "Not assigned") if hasattr(issue, 'assigned_to') else "Not assigned"
                },
                "project": {
                    "id": self._safe_get_attr(issue.project, 'id', 0),
                    "name": self._safe_get_attr(issue.project, 'name', "No project")
                }
            }
        except Exception as e:
            self.logger.error(f"❌ Error formatting issue {issue.id}: {str(e)}")
            raise
    
    def _format_comment(self, analysis: str, docs_count: int) -> str:
        """Форматирует комментарий для Redmine без эмодзи"""
        # Удаляем эмодзи из анализа (на всякий случай)
        clean_analysis = _remove_emojis(analysis)
        
        return (
            "AI Analysis (RAG):\n"
            "-----------------\n"
            f"{clean_analysis}\n\n"
            f"Documents used: {docs_count}\n"
            "-----------------\n"
            f"Timestamp: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}"
        )
    
    def _safe_get_attr(self, obj, attr, default=""):
        """Безопасное получение атрибута объекта"""
        try:
            if hasattr(obj, attr):
                return getattr(obj, attr, default)
            return default
        except ResourceAttrError:
            return default
        except Exception:
            return default

            notifier = TelegramNotifier(
                self.config.telegram_bot_token,
                self.config.telegram_chat_id,
                self.config.telegram_mapping_file
            )

                    f"Redmine processing failed: {str(e)}",
                    self.config.redmine_url
                )
    
    def run_cli(self):
        """Запуск CLI интерфейса"""
        if 'cli' not in self.channels:
            self.logger.error("❌ CLI channel not available")
            return
        
        self.channels['cli'].run_interactive()

        """Получение списка доступных провайдеров"""
        return list(self.provider_factory.providers.keys())
    
    def get_available_channels(self) -> list:
        """Получение списка доступных каналов"""
        return list(self.channels.keys())

# retrieval/reranker.py
import logging
from typing import List, Dict, Any
from sentence_transformers import CrossEncoder

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class Reranker:
    def __init__(self, model_name: str = "BAAI/bge-reranker-v2-m3", cache_folder: str = "models"):
        logger.info(f"Загрузка реранкера: {model_name}")
        self.model = CrossEncoder(
            model_name,
            cache_folder=cache_folder,
            max_length=512
        )

    def rerank(self, query: str, documents: List[Dict[str, Any]], top_k: int = 5) -> List[Dict[str, Any]]:
        if not documents:
            return []
        
        logger.info(f"Реранкинг {len(documents)} документов для запроса: {query}")
        pairs = [(query, doc["text"]) for doc in documents]
        # Уменьшаем batch_size для экономии VRAM
        scores = self.model.predict(pairs, batch_size=8)
        
        for doc, score in zip(documents, scores):
            doc["rerank_score"] = float(score)
        
        reranked = sorted(documents, key=lambda x: x["rerank_score"], reverse=True)
        logger.info(f"Лучший score после реранкинга: {reranked[0]['rerank_score']:.4f}")
        return reranked[:top_k]

        )
        self._create_collection()

    def _create_collection(self):
        if not self.client.collection_exists(self.collection_name):
            self.client.create_collection(
                collection_name=self.collection_name,
                vectors_config=VectorParams(size=1024, distance=Distance.COSINE)
            )
            logger.info(f"Коллекция создана: {self.collection_name}")
        else:
            logger.info(f"Коллекция уже существует: {self.collection_name}")

    def embed_texts(self, texts: List[str]) -> List[List[float]]:
        logger.info(f"Генерация эмбеддингов для {len(texts)} текстов в {self.collection_name}...")
        embeddings = self.embedding_model.encode(
            texts,
            batch_size=32,
            show_progress_bar=False,
            convert_to_numpy=True,
            normalize_embeddings=True
        )
        return embeddings.tolist()

    def add_documents(self, documents: List[Dict[str, Any]], replace: bool = True, batch_size: int = 100):
        if not documents:
            logger.warning("Нет документов для добавления")
            return
        
        if replace:
            if self.client.collection_exists(self.collection_name):
                logger.info(f"Удаление старой коллекции: {self.collection_name}")
                self.client.delete_collection(self.collection_name)
            self._create_collection()

        texts = [doc["text"] for doc in documents]
        embeddings = self.embed_texts(texts)
        
        points = []
        for doc, embedding in zip(documents, embeddings):
            doc_id = hashlib.md5(f"{doc['text']}{doc['metadata']['source']}".encode()).hexdigest()
            points.append(
                PointStruct(
                    id=doc_id,
                    vector=embedding,
                    payload={
                        "text": doc["text"],
                        "metadata": {**doc["metadata"], "source_db": self.collection_name}
                    }
                )
            )

        total = len(points)
        logger.info(f"Добавление {total} точек в {self.collection_name} пакетами по {batch_size}...")
        for i in range(0, total, batch_size):
            batch = points[i:i + batch_size]
            self.client.upsert(collection_name=self.collection_name, points=batch)
            logger.info(f"  [{self.collection_name}] Отправлено: {min(i + batch_size, total)}/{total}")
        
        logger.info(f"✅ Документы успешно добавлены в {self.collection_name}")

    def search(self, query: str, top_k: int = 20) -> List[Dict[str, Any]]:
        if not isinstance(query, str) or not query.strip():
            logger.warning("Получен пустой или некорректный запрос")
            return []
        
        clean_query = query.strip()
        try:
            query_embedding = self.embedding_model.encode(
                [clean_query],
                convert_to_numpy=True,
                normalize_embeddings=True
            )[0].tolist()
        except Exception as e:
            logger.error(f"Ошибка при генерации эмбеддинга для '{clean_query}': {e}")
            return []

        results = self.client.query_points(
            collection_name=self.collection_name,
            query=query_embedding,
            limit=top_k
        )
        
        return [{
            "text": hit.payload["text"],
            "metadata": hit.payload["metadata"],
            "score": hit.score,
            "source_db": self.collection_name
        } for hit in results.points]

    def process_issues(self):
        """Основной метод обработки задач из Redmine"""
        self.logger.info("🔄 Starting Redmine issue processing...")
        
        try:
            issues = self.redmine_client.find_target_issues(self.config.target_status)
            
            if not issues:
                self.logger.info("📭 No issues found in target status")
                return
            
            self.logger.info(f"📋 Found {len(issues)} issues to process")
            
            for issue in issues:
                self._process_single_issue(issue)
                
        except Exception as e:
            self.logger.error(f"❌ Error processing Redmine issues: {e}", exc_info=True)
            # Используем существующую систему уведомлений
            if self.core.telegram_notifier:
                self.core.telegram_notifier.send_error_notification(
                    {"Заголовок": "Redmine Processing Error"}, 
                    str(e), 
                    self.config.redmine_url
                )
    
    def _process_single_issue(self, issue: Dict[str, Any]):
        """Обрабатывает одну задачу из Redmine"""
        issue_info = self.redmine_client.get_issue_display_info(issue)
        self.logger.info(f"⚙️ Processing issue: {issue_info}")
        
        try:
            # Формируем запрос из описания задачи
            query = issue.get('description', '').strip()
            if not query:
                query = issue.get('subject', '')
            
            if not query:
                self.logger.warning(f"⚠️ Issue #{issue['id']} has no description or title")
                return
            
            # Получаем провайдера по умолчанию для Redmine
            provider = self.core.provider_factory.get_default_provider('redmine')
            
            # Получаем контекст (только для RAG провайдера)
            context = provider.get_context(query) if hasattr(provider, 'get_context') else None
            
            # Генерируем ответ
            answer = provider.generate_answer(query, context)
            
            # Получаем источники (только для RAG)
            sources = provider.get_sources(context) if hasattr(provider, 'get_sources') and context else []
            
            # Сохраняем в историю
            query_id = self.core.storage.save_query(
                channel="redmine",
                provider=provider.get_provider_name(),
                query=query,
                answer=answer,
                context=context,
                sources=sources,
                issue_id=issue['id'],
                project_name=issue.get('project', {}).get('name', '')
            )
            
            # Формируем комментарий с источниками для RAG
            full_comment = self._format_comment(answer, sources, provider.get_provider_name())
            
            # Добавляем комментарий в Redmine
            if self.redmine_client.add_analysis_comment(issue['id'], full_comment, len(sources or [])):
                self.logger.info(f"✅ Analysis comment added to issue #{issue['id']}")
                
                # Отправляем уведомление в Telegram (используем существующую логику)
                if self.core.telegram_notifier:
                    self.core.telegram_notifier.send_success_notification(issue, self.config.redmine_url)
            else:
                self.logger.error(f"❌ Failed to add comment to issue #{issue['id']}")
                
        except Exception as e:
            self.logger.error(f"❌ Error processing issue #{issue['id']}: {e}", exc_info=True)

import psycopg2
from psycopg2 import sql
from typing import Dict, List, Any, Optional, Union
import json
import logging
from datetime import datetime

class PostgreSQLStorage:
    """Хранилище для истории запросов/ответов в PostgreSQL с pgvector"""
    
    def __init__(self, config):
        self.logger = logging.getLogger(__name__)
        self.config = config
        self.conn = None
        self._connect()
        self._create_tables()
        self.logger.info("✅ PostgreSQL storage initialized successfully")
    
    def _connect(self):
        """Устанавливает соединение с PostgreSQL"""
        try:
            self.conn = psycopg2.connect(
                host=self.config.pg_host,
                port=self.config.pg_port,
                database=self.config.pg_db,
                user=self.config.pg_user,
                password=self.config.pg_password
            )
            self.logger.info("✅ Connected to PostgreSQL database")
        except Exception as e:
            self.logger.error(f"❌ Failed to connect to PostgreSQL: {e}")
            raise
    
    def _create_tables(self):
        """Создает необходимые таблицы в базе данных"""
        with self.conn.cursor() as cur:
            # Основная таблица истории запросов
            cur.execute("""
            CREATE TABLE IF NOT EXISTS query_history (
                id SERIAL PRIMARY KEY,
                timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                channel VARCHAR(50) NOT NULL,
                provider VARCHAR(50) NOT NULL,
                query TEXT NOT NULL,
                context JSONB,
                answer TEXT NOT NULL,
                sources JSONB,
                execution_time FLOAT,
                issue_id INTEGER,
                project_name VARCHAR(255),
                user_id VARCHAR(100),
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
            """)
            
            # Таблица для аналитики качества ответов
            cur.execute("""
            CREATE TABLE IF NOT EXISTS response_feedback (
                id SERIAL PRIMARY KEY,
                query_id INTEGER REFERENCES query_history(id) ON DELETE CASCADE,
                rating INTEGER CHECK (rating BETWEEN 1 AND 5),
                feedback_text TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
            """)
            
            # Индексы для быстрого поиска
            cur.execute("""
            CREATE INDEX IF NOT EXISTS idx_query_history_channel 
            ON query_history(channel, timestamp DESC)
            """)
            
            cur.execute("""
            CREATE INDEX IF NOT EXISTS idx_query_history_provider 
            ON query_history(provider, timestamp DESC)
            """)
            
            cur.execute("""
            CREATE INDEX IF NOT EXISTS idx_query_history_timestamp 
            ON query_history(timestamp DESC)
            """)
            
            self.conn.commit()
    
    def save_query(self, channel: str, provider: str, query: str, 
                  answer: str, context: Optional[Dict] = None, sources: Optional[List[Dict]] = None,
                  execution_time: Optional[float] = None, issue_id: Optional[int] = None,
                  project_name: Optional[str] = None, user_id: Optional[str] = None) -> int:
        """Сохраняет запрос и ответ в базу"""
        with self.conn.cursor() as cur:
            cur.execute("""
            INSERT INTO query_history 
            (channel, provider, query, context, answer, sources, execution_time, issue_id, project_name, user_id)
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, %s)
            RETURNING id
            """, (
                channel,
                provider,
                query,
                json.dumps(context) if context else None,
                answer,
                json.dumps(sources) if sources else None,
                execution_time,
                issue_id,
                project_name,
                user_id
            ))
            query_id = cur.fetchone()[0]
            self.conn.commit()
            return query_id
    
    def get_query_history(self, limit: int = 100, channel: Optional[str] = None, 
                         provider: Optional[str] = None, days: int = 7) -> List[Dict]:
        """Получает историю запросов за последние N дней"""
        with self.conn.cursor() as cur:
            query = """
            SELECT id, timestamp, channel, provider, query, answer, sources, issue_id, project_name
            FROM query_history
            WHERE timestamp >= NOW() - INTERVAL '%s days'
            """
            params = [days]
            
            if channel:
                query += " AND channel = %s"
                params.append(channel)
            
            if provider:
                query += " AND provider = %s"
                params.append(provider)
            
            query += " ORDER BY timestamp DESC LIMIT %s"
            params.append(limit)
            
            cur.execute(query, tuple(params))
            rows = cur.fetchall()
            
            return [
                {
                    "id": row[0],
                    "timestamp": row[1],
                    "channel": row[2],
                    "provider": row[3],
                    "query": row[4],
                    "answer": row[5],
                    "sources": json.loads(row[6]) if row[6] else None,
                    "issue_id": row[7],
                    "project_name": row[8]
                }
                for row in rows
            ]
    
    def add_feedback(self, query_id: int, rating: int, feedback_text: Optional[str] = None) -> bool:
        """Добавляет обратную связь по ответу"""
        with self.conn.cursor() as cur:
            cur.execute("""
            INSERT INTO response_feedback (query_id, rating, feedback_text)
            VALUES (%s, %s, %s)
            """, (query_id, rating, feedback_text))
            self.conn.commit()
            return True
    
    def close(self):
        """Закрывает соединение с базой"""
        if self.conn:
            self.conn.close()
            self.logger.info("✅ PostgreSQL connection closed")
